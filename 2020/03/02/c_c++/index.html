    一. 预处理

    1. 文件包含

    预处理指令#include用于包含头文件，有两种形式：#include <xxx.h>，#include “xxx.h”。 

    #include <xxx.h>                  //标准或系统提供的头文件
#include "xxx.h"                  //用户自己创造的头文件

    后者“ ”中可以指出文件路径和文件名。如果在双引号中没有给出绝对路径，则默认为用户当前目录中的文件，此时系统首先在用户当前目录中寻找要包含的文件，若找不到再在系统目录中查找。

    2. 条件编译

    #if  表达式                             //表达式值为1便编译下面代码
#ifdef xxx                             //xxx已被宏定义就编译下面代码
#ifndef xxx                            //xxx未被宏定义就编译下面代码
#else                                  //作为其它预处理的剩余选项进行编译
#elif  表达式                           //这是一种#else和#if的组合选项只能与#if连用
#endif                                  //结束编译块的控制

    eg：

    //you.h
#include <stdio.h>
void is_tool()
{
    printf("hello world");
}


//guang.h
#include "you.h"


//main.cpp
#include "you.h"
#include "guang.h"
int main()
{
    is_tool();
    return 0;
}

    将上述代码进行编译链接后就相当于下面的代码

    #include <stdio.h>
void is_tool()
{
    printf("hello world");
}

#include <stdio.h>
void is_tool()
{
    printf("hello world");
}

int main()
{
    is_tool();
    return 0;
}

    可以看到is_tool()被重复定义了，于是编译器报错。

    将you.h换做以下代码后

    //you.h
#ifndef __you_h__
#define __you_h__
#include <stdio.h>
void is_tool()
{
    printf("hello world");
}
#endif // !__you_h__

    整体代码便是

    #ifndef __you_h__
#define __you_h__
#include <stdio.h>
void is_tool()
{
    printf("hello world");
}
#endif // !__you_h__

#ifndef __you_h__                   //由于上面代码中以定义宏__you_h__所以不编译#endif前代码
#define __you_h__
#include <stdio.h>
void is_tool()
{
    printf("hello world");
}
#endif // !__you_h__

int main()
{
    is_tool();
    return 0;
}

    即可成功编译运行

    3. 宏替换

    #define xxx yyy                           //在编译中代码里的xxx都会被yyy替代
eg：
        1.#define xxx int
        xxx y;              //int y

        2.#define xxx "hhh"
        printf(xxx);                       //printf("hhh");

        3.#define xxx {return 0;\               //宏中换行为\+回车
        }
        int main()
        xxx                          //int main(){return 0;}

#define 宏函数名(形参) 关于形参的式子
eg:
        1.#define fun(x) x*x
        fun(x+y)                //x+y*y+x宏仅会替代

        2.#define fun(x) No##x   //宏函数代替标识符时要想其形参被带入标识符要在其前面加##
        fun(1)                    //No1

        3.#define fun(x) printf(#x)   //#将形参变成字符串，而@#则将其变为字符
        fun(hhh)               //printf("hhh")

        4.#define fun(x) x*fun(x)       //产生循环则x*fun(x)里fun(x)不会被替换
        fun(1)                     //1*fun(1)

        5.#define fun(x,y...) printf(x,##y)  //##可消除因空参数产生的多余','
        fun("hhh")  //printf("hhh")
